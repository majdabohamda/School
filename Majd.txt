    const container = document.getElementById('questionsContainer');

    quizData.forEach((item, qIndex) => {
        const card = document.createElement('div');
        card.className = 'question-card';
        card.id = `q-card-${qIndex}`;

        let optionsHtml = `<span class="question-text">${item.q}</span>`;
        item.options.forEach((opt, oIndex) => {
            optionsHtml += `
                <label class="option" id="opt-label-${qIndex}-${oIndex}">
                    <input type="radio" name="q${qIndex}" value="${oIndex}">
                    <span>${opt}</span>
                </label>
            `;
        });

        card.innerHTML = optionsHtml;
        container.appendChild(card);
    });

    function calculateScore() {
        let score = 0;
        quizData.forEach((item, qIndex) => {
            const selected = document.querySelector(`input[name="q${qIndex}"]:checked`);
            const correctOIndex = item.correct;

            // Reset labels styles
            for(let i=0; i<4; i++) {
                document.getElementById(`opt-label-${qIndex}-${i}`).className = 'option';
            }

            // Mark correct answer
            document.getElementById(`opt-label-${qIndex}-${correctOIndex}`).classList.add('correct-opt');

            if (selected) {
                const userChoice = parseInt(selected.value);
                if (userChoice === correctOIndex) {
                    score++;
                } else {
                    document.getElementById(`opt-label-${qIndex}-${userChoice}`).classList.add('wrong-opt');
                }
            }
        });

        const resultBox = document.getElementById('result');
        resultBox.style.display = 'block';
        resultBox.innerText = `درجتك النهائية: ${score} من ${quizData.length}`;
        window.scrollTo({ top: 0, behavior: 'smooth' });
    }
